\textbf{Surface Integrals:}

\[\vec{G}(u,v)=(x(u,v),y(u,v),z(u,v)),\quad \begin{cases}
    \vec{T}_u(P) = \pd{\vec{G}}{u} = (\pd{x}{u},\pd{y}{u},\pd{z}{u}) \\
    \vec{T}_v(P) = \pd{\vec{G}}{v} = (\pd{x}{v},\pd{y}{v},\pd{z}{v})
\end{cases},\quad
\vec{N} = \vec{T}_u\times\vec{T}_v\]


\[\iint_S f(x,y,z)dS = \iint_D f(\vec{G}(u,v))|\vec{N}(u,v)| dudv, 
\quad dS = |\vec{N}|dudv\]

\[\iint_S \vec{F}\cdot d\vec{S} = \iint_S \vec{F}\cdot\vec{n} dS = \iint_S \vec{F}(\vec{G}(u,v))\cdot\vec{N}(u,v) dudv, 
\quad d\vec{S}=\vec{n}dS=\frac{\vec{N}}{|\vec{N}|}dS\]

Specially, 
\[\vec{G}=(x,y,g(x,y)),
\vec{N} = (-g_x, -g_y, 1),
