\AddEverypageHook{%
    \ifodd\thepage
        % Odd page (right page): vertical text on far right
        \begin{textblock}{1}(98, 5)  % 98% from left, 5% from top
            \rotatebox{90}{%
                \tiny\sffamily\textcolor{gray!50}{%
                    \thepage\ --- REAL MARGIN (RIGHT) --- \rightmark
                }%
            }%
        \end{textblock}
        % Also on far left for odd pages (spine side gets chapter info)
        \begin{textblock}{1}(1, 5)  % 1% from left
            \rotatebox{90}{%
                \tiny\sffamily\textcolor{gray!30}{%
                    \leftmark
                }%
            }%
        \end{textblock}
    \else
        % Even page (left page): vertical text on far left
        \begin{textblock}{1}(1, 5)  % 1% from left
            \rotatebox{90}{%
                \tiny\sffamily\textcolor{gray!50}{%
                    \thepage\ --- REAL MARGIN (LEFT) --- \leftmark
                }%
            }%
        \end{textblock}
        % Also on far right for even pages
        \begin{textblock}{1}(98, 5)
            \rotatebox{90}{%
                \tiny\sffamily\textcolor{gray!30}{%
                    \rightmark
                }%
            }%
        \end{textblock}
    \fi
}
